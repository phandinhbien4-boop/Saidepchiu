local Fluent = loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua"))()

local G = getgenv and getgenv() or _G
local set_readonly = setreadonly or make_writeable
local hook_metamethod = hookmetamethod or function() end
local new_cclosure = newcclosure or function(f) return f end

task.spawn(function()
    if not hook_metamethod then return end
    
    local mt = getrawmetatable(game)
    set_readonly(mt, false)
    local old_namecall = mt.__namecall
    local old_index = mt.__index

    mt.__namecall = new_cclosure(function(self, ...)
        local method = getnamecallmethod()
        local args = {...}
        
        if method == "FireServer" then
            if self.Name == "RemoteEvent" or self.Name:find("Check") or self.Name:find("Admin") or self.Name:find("Detection") then
                return nil
            end
            if args[1] == "WalkSpeed" or args[1] == "JumpPower" or args[1] == "Teleport" then
                return nil
            end
        end
        
        if method == "Kick" then 
            return nil 
        end
        
        return old_namecall(self, ...)
    end)

    mt.__index = new_cclosure(function(self, key)
        if not checkcaller() and self:IsA("Humanoid") then
            if key == "WalkSpeed" then return 16 end
            if key == "JumpPower" then return 50 end
        end
        return old_index(self, key)
    end)
    
    set_readonly(mt, true)
end)
