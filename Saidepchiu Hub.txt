local function CreateCustomToggle(parent, title, desc, callback)
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(1, -20, 0, 60)
    frame.Position = UDim2.new(0, 10, 0, 0)
    frame.BackgroundTransparency = 1
    frame.Parent = parent
    
    local toggleBtn = Instance.new("ImageButton")
    toggleBtn.Name = "Toggle"
    toggleBtn.Image = "rbxassetid://734800536842"
    toggleBtn.ImageRectOffset = Vector2.new(0, 0)
    toggleBtn.ImageRectSize = Vector2.new(50, 50)
    toggleBtn.Size = UDim2.new(0, 50, 0, 50)
    toggleBtn.Position = UDim2.new(0, 0, 0.5, -25)
    toggleBtn.BackgroundTransparency = 1
    toggleBtn.Parent = frame
    
    local titleLabel = Instance.new("TextLabel")
    titleLabel.Text = title
    titleLabel.TextColor3 = Color3.new(1, 1, 1)
    titleLabel.TextSize = 18
    titleLabel.Font = Enum.Font.GothamBold
    titleLabel.TextXAlignment = Enum.TextXAlignment.Left
    titleLabel.Size = UDim2.new(1, -60, 0, 25)
    titleLabel.Position = UDim2.new(0, 60, 0, 5)
    titleLabel.BackgroundTransparency = 1
    titleLabel.Parent = frame
    
    local descLabel = Instance.new("TextLabel")
    descLabel.Text = desc
    descLabel.TextColor3 = Color3.new(0.7, 0.7, 0.7)
    descLabel.TextSize = 14
    descLabel.Font = Enum.Font.Gotham
    descLabel.TextXAlignment = Enum.TextXAlignment.Left
    descLabel.Size = UDim2.new(1, -60, 0, 20)
    descLabel.Position = UDim2.new(0, 60, 0, 30)
    descLabel.BackgroundTransparency = 1
    descLabel.Parent = frame
    
    local enabled = false
    toggleBtn.MouseButton1Click:Connect(function()
        enabled = not enabled
        if enabled then
            toggleBtn.ImageRectOffset = Vector2.new(50, 0)
        else
            toggleBtn.ImageRectOffset = Vector2.new(0, 0)
        end
        callback(enabled)
    end)
    
    return toggleBtn
end
