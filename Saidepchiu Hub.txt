local Fluent = loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua"))()
local Player, WS, RS = game:GetService("Players").LocalPlayer, game:GetService("Workspace"), game:GetService("RunService")
local VU, CG = game:GetService("VirtualUser"), game:GetService("CoreGui")

-- Anti-Kick & Fake Ping
local anti; anti = hookmetamethod(game, "__namecall", function(s, ...)
    if getnamecallmethod() == "Kick" then return end return anti(s, ...)
end)
RS.Heartbeat:Connect(function() pcall(function() game:GetService("Stats").Network.ServerStatsItem["Data Ping"]:SetValue(math.random(45, 85)) end) end)

-- UI Toggle Button
local ScreenGui = Instance.new("ScreenGui", CG)
local ToggleBtn = Instance.new("TextButton", ScreenGui)
ToggleBtn.Size, ToggleBtn.Position, ToggleBtn.BackgroundColor3, ToggleBtn.Text = UDim2.new(0, 60, 0, 60), UDim2.new(0, 15, 0.5, -30), Color3.fromRGB(0, 255, 127), "ON"
ToggleBtn.Draggable = true
Instance.new("UICorner", ToggleBtn).CornerRadius = UDim.new(0, 20)

-- Main Menu
local Window = Fluent:CreateWindow({ Title = "Saidepchiu Hub", SubTitle = "Blox Fruits 2026", TabWidth = 160, Size = UDim2.fromOffset(600, 500), Acrylic = true, Theme = "Dark" })
local Tabs = { Farm = Window:AddTab({ Title = "Farm", Icon = "swords" }), ESP = Window:AddTab({ Title = "ESP", Icon = "eye" }), Teleport = Window:AddTab({ Title = "Teleport", Icon = "map-pin" }) }
local Options = Fluent.Options

-- Functions
local function attack() pcall(function() VU:ClickButton1(Vector2.new(), WS.CurrentCamera.CFrame) end) end
local function to(pos) pcall(function() Player.Character.HumanoidRootPart.CFrame = CFrame.new(pos.X, pos.Y + 50, pos.Z) end) end

-- Data (Rút gọn danh sách quái siêu nén)
local mobs = {[1]="Bandit",[30]="Monkey",[60]="Pirate",[90]="Brute",[120]="Desert Officer",[150]="Snow Bandit",[180]="Vampire",[1500]="Awakened Raider",[2800]="Tyrant Lord"}
local bosses = {"The Gorilla King","Bobby","Jeremy","Diamond","Saber Expert","Greybeard","Rip_Indra","Dough King","Unbound Werewolf"}

local function getMob()
    local lv, res = Player.Data.Level.Value, "Bandit"
    for l, m in pairs(mobs) do if lv >= l then res = m end end return res
end

-- Farm Logic
local function autoFarm(name, isBoss)
    for _, v in pairs(WS.Enemies:GetChildren()) do
        if v.Name == name and v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
            to(v.HumanoidRootPart.Position)
            attack() return true
        end
    end
end

-- UI Elements
Tabs.Farm:AddSection("⚔️ AUTO FARM")
Tabs.Farm:AddToggle("FarmLevel", {Title = "Auto Farm Level"}):OnChanged(function(v) _G.FL = v end)
Tabs.Farm:AddToggle("FarmAllBoss", {Title = "Auto Farm All Boss"}):OnChanged(function(v) _G.AB = v end)
Tabs.Farm:AddDropdown("BossSel", {Title = "Chọn Boss", Values = bosses, Default = 1})
Tabs.Farm:AddToggle("FarmBoss", {Title = "Farm Boss đã chọn"}):OnChanged(function(v) _G.FB = v end)

-- ESP Logic
Tabs.ESP:AddToggle("ESPPlr", {Title = "ESP Player"}):OnChanged(function(v) _G.EP = v end)

-- Teleport Data
local islands = {["Jungle"] = Vector3.new(1600, 20, 50), ["Cafe"] = Vector3.new(-450, 30, -3000), ["Mansion"] = Vector3.new(-3000, 50, -7000)}
Tabs.Teleport:AddDropdown("Isle", {Title = "Chọn đảo", Values = {"Jungle", "Cafe", "Mansion"}})
Tabs.Teleport:AddButton({Title = "Teleport", Callback = function() if islands[Options.Isle.Value] then to(islands[Options.Isle.Value]) end end})

-- Main Loop (Trái tim của Script)
task.spawn(function()
    while task.wait() do
        if _G.FL then autoFarm(getMob()) end
        if _G.AB then for _, b in pairs(bosses) do if autoFarm(b) then break end end end
        if _G.FB then autoFarm(Options.BossSel.Value) end
    end
end)

ToggleBtn.MouseButton1Click:Connect(function() Window:Minimize() ToggleBtn.Text = (ToggleBtn.Text == "ON") and "OFF" or "ON" end)
Window:SelectTab(1)
